import psycopg2
from flask import Flask, render_template
from flask_cors import CORS

#crear la aplicacion de flask
app= Flask(__name__)

#configurar CORS para permitir todos los origenes
CORS(app)

def get_db_connection():
	conn = psycopg2.connect(
		host = "",
		database = "",
		user = "",
		password = ""
	)
	return conn
#raiz que devuelve un mensaje
@app.route('/')
def home():
	print("Hola, mundo")
	return render_template("index.html")

if __name__== "__main__":
	app.run(debug=True)
